func isValid(s string) bool {
	m := map[rune]rune{
		')': '(',
		'}': '{',
		']': '[',
	}

	stack := make([]rune, 0, len(s))
	for _, ch := range s {
		if _, ok := m[ch]; ok { // 右括号
			if len(stack) == 0 {
				return false
			}
			top := stack[len(stack)-1]
			if v, ok := m[ch]; !ok || v != top {
				return false
			}
			stack = stack[:len(stack)-1] // pop
		} else { // 左括号
			stack = append(stack, ch)
		}
	}
	return len(stack) == 0
}
