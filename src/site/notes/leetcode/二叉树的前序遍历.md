## 递归

```go
func preorderTraversal(root *TreeNode) []int {
	var vals []int

	var preorder func(root *TreeNode)
	// 通过闭包的方式来引用方法内的 vals
	// 这样就能实现所有的递归函数都能共享 vals 变量
	preorder = func(root *TreeNode) {
		if root == nil {
			return
		}
		vals = append(vals, root.Val)
		preorder(root.Left)
		preorder(root.Right)
	}
	preorder(root)
	return vals
}
```
