## 递归

```go
func inorderTraversal(root *TreeNode) []int {
    var vals []int
	var inorder func(root *TreeNode)
	inorder = func(root *TreeNode) {
		if root == nil {
			return
		}
		inorder(root.Left)
		vals = append(vals, root.Val)
		inorder(root.Right)
	}
	inorder(root)
	return vals
    
}
```
