关键是通过 *carry* 变量来**记录进位**。

思路：
1. 同时遍历两个链表，如果其中为空，则用 0 代替；
2. 当前位的总和：*l1.val  = l2.val + carry*
3. 注意处理**最后 carry > 0** 的情况，即多出了一位。

## Go

```go
func addTwoNumbers(l1 *ListNode, l2 *ListNode) *ListNode {
	res := &ListNode{}
	temp := 0
	cur := res
	for l1 != nil || l2 != nil {
		sum := 0
		if l1 != nil {
			sum += l1.Val
			l1 = l1.Next
		}
		if l2 != nil {
			sum += l2.Val
			l2 = l2.Next
		}
		cur.Next = &ListNode{Val: (sum + temp) % 10}
		temp = (sum + temp) / 10
		cur = cur.Next
	}
	if temp > 0 {
		cur.Next = &ListNode{Val: temp}
	}
	return res.Next
}
```

## Java

```java
public ListNode addTwoNumbers(ListNode l1, ListNode l2) {  
    int carry = 0;  
    ListNode c1 = l1, c2 = l2;  
    ListNode dummy = new ListNode();  
    ListNode cur = dummy;  
  
    while (c1 != null || c2 != null) {  
        int v1 = c1 != null ? c1.val : 0;  
        int v2 = c2 != null ? c2.val : 0;  
        int sum = v1 + v2 + carry;  
  
        cur.next = new ListNode(sum % 10);  
        carry = sum / 10;  
  		
		// 记得判断后再向前遍历，因为此时可能有一个链表已经停止
        if (c1 != null) c1 = c1.next;  
        if (c2 != null) c2 = c2.next;  
        cur = cur.next;  
    }
	// 处理进位问题(多出一位)
    if (carry > 0) {  
        cur.next = new ListNode(carry);
    }  
    return dummy.next;  
}
```
