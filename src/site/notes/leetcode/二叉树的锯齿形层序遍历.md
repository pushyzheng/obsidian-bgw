## 思路

使用[[leetcode/二叉树的层序遍历\|二叉树的层序遍历]]，区别在于当从右到左遍历时，反向添加到 level 每层结果的数组当中即可。

## 代码

```go
func zigzagLevelOrder(root *TreeNode) [][]int {
	if root == nil {
		return nil
	}
	queue := []*TreeNode{root}
	var flag bool
	var res [][]int
	for len(queue) > 0 {
		level := make([]int, len(queue))
		n := len(queue)
		for i := 0; i < n; i++ {
			node := queue[0]
			queue = queue[1:]
			if flag {
				level[n-i-1] = node.Val  // 从右到左遍历, 反向添加
			} else {
				level[i] = node.Val      // 从左到右遍历, 正向添加
			}
			if node.Left != nil {
				queue = append(queue, node.Left)
			}
			if node.Right != nil {
				queue = append(queue, node.Right)
			}
		}
		flag = !flag
		res = append(res, level)
	}
	return res
}
```
