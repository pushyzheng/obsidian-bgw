定义小于 x 的链表 *less* 和大于 x 的链表 *more*，直接遍历链表：
- 小于 x 的节点拼接到 *less* 中
- 大于 x 的节点拼接到 *more* 中

最后再将 *less* 和 *more* 链表拼接即可。

参考代码：

## Go

```go
func partition(head *ListNode, x int) *ListNode {
	less, more := &ListNode{}, &ListNode{}
	lc, mc := less, more
	for cur := head; cur != nil; cur = cur.Next {
		if cur.Val < x {
			lc.Next = cur
			lc = lc.Next
		} else {
			mc.Next = cur
			mc = mc.Next
		}
	}
	// 这一步至关重要，可能存在后边的元素是小于 x 的
	// 但是由于要保持链表中的相对位置，所以需要把 Next 置空
	mc.Next = nil 
	lc.Next = more.Next
	return less.Next
}
```

## Java

```java
public ListNode partition(ListNode head, int x) {
	if (head == null || head.next == null) {
		return head;
	}

	ListNode lessDummy = new ListNode(-1);
	ListNode less = lessDummy;

	ListNode moreDummy = new ListNode(-1);
	ListNode more = moreDummy;

	while (head != null) {
		if (head.val < x) {
			less.next = head;
			less = less.next;
		} else {
			more.next = head;
			more = more.next;
		}
		head = head.next;
	}
	more.next = null;
	less.next = moreDummy.next;
	return lessDummy.next;
}
```
